### cleo.configure()
Configure the Cleo instance with global options.

```typescript
interface CleoConfig {
  broker: BrokerConfig;
  workers: WorkerConfig;
  monitoring?: MonitoringConfig;
  logging?: LoggingConfig;
  plugins?: PluginConfig[];
}

// Example usage
cleo.configure({
  broker: {
    type: 'redis',
    url: process.env.REDIS_URL,
    options: {
      tls: true,
      maxRetriesPerRequest: 3
    }
  },
  workers: {
    concurrency: 5,
    maxMemory: '1GB',
    gracefulShutdown: true
  },
  monitoring: {
    enabled: true,
    metrics: true,
    dashboard: {
      port: 3000
    }
  },
  logging: {
    level: 'info',
    format: 'json'
  }
});
```

### @cleo.task Decorator
Define a task with various options.

```typescript
interface TaskOptions {
  name?: string;
  queue?: string;
  priority?: TaskPriority;
  retries?: number;
  timeout?: number | string;
  backoff?: BackoffOptions;

  dependencies?: string[];
```

### Task Enqueuing Methods

#### cleo.enqueue()
Enqueue a single task.

```typescript
interface EnqueueOptions {
  delay?: number | string;
  priority?: TaskPriority;
  idempotencyKey?: string;
  context?: Record<string, any>;
}

// Example usage
const taskId = await cleo.enqueue('process-order', '12345', {
  delay: '5m',
  priority: TaskPriority.HIGH
});

logger.info('Task enqueued', {
  fileName: 'api/enqueue.ts',
  lineNo: 75,
  functionName: 'enqueueOrder',
  variable: 'taskId',
  value: taskId
});
```

#### cleo.enqueueBatch()
Enqueue multiple tasks at once.

```typescript
interface BatchTask {
  name: string;
  data: any[];
  options?: EnqueueOptions;
}

// Example usage
const tasks: BatchTask[] = orders.map(order => ({
  name: 'process-order',
  data: [order.id],
  options: { priority: TaskPriority.NORMAL }
}));

const results = await cleo.enqueueBatch(tasks);
```
